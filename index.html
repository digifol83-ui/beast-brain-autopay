Quick Start

Initialize Transak with your API key and configuration.

const transak = new Transak({
  apiKey: 'YOUR_API_KEY',

  network: 'solana',
  walletAddress: 'RECIPIENT_WALLET',
});

transak.init();

Backend: PaymentIntent

Create a payment intent for the lockbox flow.

python 
import stripe
stripe.api_key = "sk_live_..."

def create_lockbox_payment(amount_usd, metadata):
    intent = stripe.PaymentIntent.create(
        amount=int(amount_usd * 100),
        currency="usd",
        metadata=metadata,
        automatic_payment_methods={"enabled": True},
    )
    return intent.client_secret

@app.post("/webhook/stripe")
async def stripe_webhook(request: Request):
    payload = await request.body()
    event = stripe.Webhook.construct_event(
        payload, request.headers['stripe-signature'],
        'whsec_...')
    if event.type == 'payment_intent.succeeded':
        await trigger_crypto_payout(event)

Create Invoice

Generate a crypto invoice via REST API.

python 
import requests

def create_plisio_invoice(amount_usd, order_id):
    res = requests.post('https://plisio.net/api/v1/invoices/new',
        params={'api_key': 'YOUR_API_KEY'},
        json={
            'order_name': 'BeastBrain AutoPay',
            'order_number': order_id,
            'source_amount': amount_usd,
            'source_currency': 'USD',
            'currency': 'USDC',
        })
    return res.json()['data']['invoice_url']

Create Exchange

Initiate fiat-to-crypto or crypto-to-crypto swap.

import requests

def create_changenow_swap(amount, from_, to, address):
    res = requests.post(
        'https://api.changenow.io/v2/exchange',
        headers={'x-api-key': 'YOUR_API_KEY'},
        json={'from': from_, 'to': to,
              'amount': amount, 'address': address})
    return res.json()

Secret Manager

Securely store and retrieve API keys at runtime.

python 
from google.cloud import secretmanager

client = secretmanager.SecretManagerServiceClient()

def get_secret(secret_id):
    name = f'projects/PROJECT/secrets/{secret_id}/versions/latest'
    resp = client.access_secret_version(request={'name': name})
    return resp.payload.data.decode('UTF-8')

Send USDC

Transfer USDC (SPL token) to any wallet.

js 
import { Connection, Keypair, PublicKey }
  from '@solana/web3.js';
import { getOrCreateAssociatedTokenAccount,
         transfer } from '@solana/spl-token';

const USDC_MINT = new PublicKey(
  'EPjFWdd5AufqSSqeM2qN1xzybapC8G4wEGGkZwyTDt1v');

const sendUSDC = async (from, to, amount) => {
  const conn = new Connection(
    'https://api.mainnet-beta.solana.com');
  const src = await getOrCreateAssociatedTokenAccount(
    conn, from, USDC_MINT, from.publicKey);
  const dest = await getOrCreateAssociatedTokenAccount(
    conn, from, USDC_MINT, new PublicKey(to));
  return await transfer(conn, from, src.address,
    dest.address, from, amount * 1_000_000);
};

Verify Transaction

Confirm on-chain settlement for payment records.

js 
const verifyTx = async (signature) => {
  const conn = new Connection(
    'https://api.mainnet-beta.solana.com');
  const tx = await conn.getTransaction(signature, {
    commitment: 'confirmed' });
  return tx && !tx.meta.err;
};

Wallet Balance

Check USDC balance of any wallet.

js 
const getUSDCBalance = async (walletAddr) => {
  const conn = new Connection(
    'https://api.mainnet-beta.solana.com');
  const tokens = await conn
    .getParsedTokenAccountsByOwner(
      new PublicKey(walletAddr),
      { mint: new PublicKey(
        'EPjFWdd5AufqSSqeM2qN1xzybapC8G4wEGGkZwyTDt1v') });
  const bal = tokens.value[0]
    ?.account.data.parsed.info.tokenAmount.uiAmount;
  return bal || 0;
};
