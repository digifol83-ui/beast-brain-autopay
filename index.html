def create_lockbox_payment(amount_usd, metadata):
    intent = stripe.PaymentIntent.create(
        amount=int(amount_usd * 100),
        currency="usd",
        metadata=metadata,
        automatic_payment_methods={"enabled": True},
    )
    return intent.client_secret

def create_plisio_invoice(amount_usd, order_id):

🔑 Create Exchange

Initiate a fiat-to-crypto or crypto-to-crypto swap.

python 
import requests

def create_changenow_swap(amount, from_, to, address):
    res = requests.post(
        'https://api.changenow.io/v2/exchange',
        headers={'x-api-key': 'YOUR_API_KEY'},
        json={
            'from': from_,       # e.g. 'usd'
            'to': to,             # e.g. 'usdcsol'
            'amount': amount,   # e.g. 100
            'address': address, # recipient wallet
        })
    return res.json()  # { id, payinAddress, status }

🔑 Secret Manager

Securely store and retrieve API keys at runtime.

python 
from google.cloud import secretmanager

client = secretmanager.SecretManagerServiceClient()

def get_secret(secret_id):
    name = f'projects/PROJECT/secrets/{secret_id}/versions/latest'
    response = client.access_secret_version(request={'name': name})
    return response.payload.data.decode('UTF-8')

# Usage
stripe_key = get_secret('stripe-api-key')
transak_key = get_secret('transak-api-key')

🔑 Send USDC

Transfer USDC (SPL token) to any wallet.

js 
import { Connection, Keypair, PublicKey }
  from '@solana/web3.js';
import { getOrCreateAssociatedTokenAccount,
         transfer } from '@solana/spl-token';

const USDC_MINT = new PublicKey(
  'EPjFWdd5AufqSSqeM2qN1xzybapC8G4wEGGkZwyTDt1v');

const sendUSDC = async (from, to, amount) => {
  const conn = new Connection('https://api.mainnet-beta.solana.com');
  const src = await getOrCreateAssociatedTokenAccount(
    conn, from, USDC_MINT, from.publicKey);
  const dest = await getOrCreateAssociatedTokenAccount(
    conn, from, USDC_MINT, new PublicKey(to));
  return await transfer(conn, from,
    src.address, dest.address, from,
    amount * 1_000_000);  // USDC has 6 decimals
};

🎯 Verify Transaction

Confirm on-chain settlement for payment records.

js 
const verifyTx = async (signature) => {
  const conn = new Connection(
    'https://api.mainnet-beta.solana.com');
  const tx = await conn.getTransaction(signature, {
    commitment: 'confirmed'
  });
  if (tx && !tx.meta.err) {
    console.log(`✅ Confirmed: {signature}`);
    return true;
  }
  return false;
};

🔄 Wallet Balance

Check USDC balance of any wallet.

js 
const getUSDCBalance = async (walletAddr) => {
  const conn = new Connection(
    'https://api.mainnet-beta.solana.com');
  const tokens = await conn.getParsedTokenAccountsByOwner(
    new PublicKey(walletAddr),
    { mint: new PublicKey(
      'EPjFWdd5AufqSSqeM2qN1xzybapC8G4wEGGkZwyTDt1v') }
  );
  const balance = tokens.value[0]
    ?.account.data.parsed.info.tokenAmount.uiAmount;
  return balance || 0;
};
